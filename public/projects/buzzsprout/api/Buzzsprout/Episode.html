def duration
  segments = [self[:duration]/60]
  segments << (self[:duration] % 60 ) 
  segments.map {|t| t.to_s.rjust(2, '0')}.join(':')
end

def duration=(value)
  new_duration = value.to_s.split(":").reverse
  s, m = new_duration
  self[:duration] = (s.to_i + (m.to_i*60))
end
